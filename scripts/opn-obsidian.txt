@echo off
setlocal

set VAULT=C:\path\to\your\vault
set OBSIDIAN=C:\path\to\Obsidian.exe

cd /d "%VAULT%"

:: Pull on open
git pull

:: Commit any pre-existing changes
call :sync

:: Open Obsidian
start "" "obsidian"

:: Wait for Obsidian to close
timeout /t 3 /nobreak >nul
:waitloop
tasklist | find /i "Obsidian.exe" >nul
if %errorlevel%==0 (
    timeout /t 5 /nobreak >nul
    goto waitloop
)

:: Commit on close
call :sync
exit /b

:: -------
:sync
git diff --quiet >nul 2>&1
if %errorlevel%==1 goto docommit
git ls-files --others --exclude-standard | findstr . >nul
if %errorlevel%==0 goto docommit
exit /b

:docommit
for /f "tokens=1-2 delims= " %%a in ("%date% %time%") do set DT=%%a_%%b
git add .
git commit -m "%DT%"
git push
exit /b
