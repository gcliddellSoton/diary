def IsBlipFreq(s, fs, STFTopts, (lb, ub), threshFreq)

1. STFTopts = {NFFT, window}
2. SFT = ShortTimeFFT(STFTopts)
3. f, Pxx = = SFT.stft(s)
4. idxs = where(lb \< f \< ub)
